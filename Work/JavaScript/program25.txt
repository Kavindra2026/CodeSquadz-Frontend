function adddata()
{
   if(jbox[0].value=="left")
   {
    myarr.unshift(jinput.value);
   }
   else if(jbox[0].value=="right")
   {
    myarr.push(jinput.value);
   }
   else
   {
    alert("please select direction to add data");
   }
   displaydata();
}


function removedata()
{
   if(jbox[0].value=="left")
   {
    myarr.shift(jinput.value);
   }
   else if(jbox[0].value=="right")
   {
    myarr.pop(jinput.value);
   }
   else
   {
    alert("please select direction to remove data");
   }
   displaydata();
}

function jsearch()
{  var resultdata;
    var searchvalue = jinput.value;
    if(jbox[0].value=="left")
    {
     resultdata = myarr.indexOf(searchvalue);   
    }
    else if(jbox[0].value=="right")
        {
         resultdata = myarr.lastIndexOf(searchvalue);   
        }
    else
    {
        alert("please provide direction of search data");
    }
    jbox[3].innerHTML=`Index Of Data: ${resultdata}`;

}

function jreverse()
{
